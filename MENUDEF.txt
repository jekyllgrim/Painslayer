Optionmenu "PK_Settings" {
	Class "PK_ModSettingsMenu"
	Title "$PKM_TITLE"
	StaticText ""
	// mod controls:
	StaticText "$PKM_SUB_CONTROLS", yellow
	StaticText ""
	Control "$PKM_UseCards", "netevent PK_UseGoldenCards"
	Control "$PKM_OpenTarot", "netevent PKCOpenBoard"
	Control "$PKM_OpenCodex", "netevent PKCOpenCodex"	
	Control "$PKM_LastWeapon", "+user3"
	StaticText ""
	// gameplay options:
	StaticText "$PKM_SUB_GAMEPLAY", yellow
	StaticText ""
	Option "$PKM_PKMovement", "pk_movement", "OnOff"
	Option "$PKM_PKMAutoJump", "pkzm_autojump", "OnOff", "pk_movement"
	Option "$PKM_AutoOpenTarot", "pk_autoOpenBoard", "OnOff"
	ScaleSlider "$PKM_QOLCATCHER", "pk_QoLCatcherDistance", 0, 96, 8, "$OPTVAL_OFF"
	Option "$PKM_ALLOWDEMON", "pk_allowDemonMorph", "OnOff"
	Option "$PKM_GIFTCARRDS", "pk_allowFreeCards", "OnOff"
	Option "$PKM_KeepBodies", "pk_keepbodies", "OnOff"
	StaticText ""
	// HUD options:
	StaticText "$PKM_SUB_HUD", yellow
	StaticText ""
	PKCrosshairOption "$HUDMNU_CROSSHAIRON", "crosshair"
	Option "$PKM_ShowCodexNotifs", "pk_CodexNotifs", "OnOff"
	Option "$PKM_ShowCardsInHUD", "pk_ShowCardsInHUD", "OnOff"
	StaticText ""
	// performance options:
	StaticText "$PKM_Sub_Performance", yellow
	StaticText ""
	Option "$PKM_ParticlesAmount", "pk_particles", "ParticlesAmount"
	ScaleSlider "$DSPLYMNU_MAXPARTICLES", "r_maxparticles", 3000, 32000, 100, ""
	//ScaleSlider "$PKM_DebrisNumber", "pk_maxdebris", 100, 3000, 100, ""
	Option "$PKM_MapStartSound", "pk_startsound", "OnOff"
	StaticText ""
	// a section where you can switch primary/secondary
	// fire modes for any weapon:
	StaticText "$PKM_FIREMODES", yellow
	StaticText ""
	Option "$CNTRLMNU_SLOT1", "pk_switch_Painkiller", 			"PKIL_MODES"
	Option "$CNTRLMNU_SLOT2", "pk_switch_ShotgunFreezer", 		"SHFR_MODES"
	Option "$CNTRLMNU_SLOT3", "pk_switch_StakeGrenade",			"STGR_MODES"
	Option "$CNTRLMNU_SLOT4", "pk_switch_MinigunRocket", 		"CGUN_MODES"
	Option "$CNTRLMNU_SLOT5", "pk_switch_ElectroDriver", 		"ELDR_MODES"
	Option "$CNTRLMNU_SLOT6", "pk_switch_RifleFlamethrower", 	"RIFT_MODES"
	Option "$CNTRLMNU_SLOT7", "pk_switch_BoltgunHeater", 		"BOHE_MODES"
	Command "$OPTMNU_DEFAULTS", "pk_switch_defaults"
}
